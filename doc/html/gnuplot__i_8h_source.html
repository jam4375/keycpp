842 //------------------------------------------------------------------------------
843 //
844 // A string tokenizer taken from http://www.sunsite.ualberta.ca/Documentation/
845 // /Gnu/libstdc++-2.90.8/html/21_strings/stringtok_std_h.txt
846 //
847 template <typename Container>
848 void stringtok (Container &container,
849  std::string const &in,
850  const char * const delimiters = " \t\n")
851 {
852  const std::string::size_type len = in.length();
853  std::string::size_type i = 0;
854 
855  while ( i < len )
856  {
857  // eat leading whitespace
858  i = in.find_first_not_of (delimiters, i);
859 
860  if (i == std::string::npos)
861  return; // nothing left but white space
862 
863  // find the end of the token
864  std::string::size_type j = in.find_first_of (delimiters, i);
865 
866  // push token
867  if (j == std::string::npos)
868  {
869  container.push_back (in.substr(i));
870  return;
871  }
872  else
873  container.push_back (in.substr(i, j-i));
874 
875  // set up for next loop
876  i = j + 1;
877  }
878 
879  return;
880 }

1754  // The popen() function shall execute the command specified by the string
